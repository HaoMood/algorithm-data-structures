class BST(__builtin__.object)
    Implementation of a binary search tree.
 
>>> bst = BST()
>>> bst[3] = 'red'
>>> bst[4] = 'blue'
>>> bst[6] = 'yellow'
>>> bst[2] = 'at'
>>> print(bst[6])
yellow
>>> print(bst[2])
at
